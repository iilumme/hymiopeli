5   import java.awt.BorderLayout;
6   import java.awt.Color;
7   import java.awt.Container;
8   import java.awt.Dimension;
9   import java.io.IOException;
10  import javax.swing.BoxLayout;
11  import javax.swing.ImageIcon;
12  import javax.swing.JFrame;
13  import javax.swing.JLabel;
14  import javax.swing.JPanel;
15  import javax.swing.WindowConstants;
16  
17  /**
18   * Graaffinen käyttöliittymä Hymiöpelille.
19   */
20  public class Kayttoliittyma implements Runnable {
21  
22      private static final int LEVEYS = 600;
23      private static final int KORKEUS = 600;
24  
25      private static final String HYMIOPELILOGO = "Images/logo.png";
26  
27      private JFrame frame;
28      private Piirtoalusta piirtoalusta;
29      private final HymioPeli hymiopeli;
30      private Apuri apuri;
31  
32      public Kayttoliittyma() {
33          this.hymiopeli = new HymioPeli(LEVEYS, LEVEYS);
34      }
35  
36      public JFrame getFrame() {
37          return frame;
38      }
39  
40      public Piirtoalusta getPiirtoalusta() {
41          return piirtoalusta;
42      }
43  
44      @Override
45      public void run() {
46          frame = new JFrame("Hymiöpeli");
47          frame.setPreferredSize(new Dimension(LEVEYS, KORKEUS));
48          frame.setResizable(false);
49          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
50  
51          luoKomponentit(frame.getContentPane());
52          setKeyListener();
53  
54          frame.pack();
55          frame.setVisible(true);
56      }
57  
58      private void luoKomponentit(Container c) {
59  
60          //Layout
61          BorderLayout cLayout = new BorderLayout();
62          c.setLayout(cLayout);
63          c.setBackground(new Color(249, 108, 57));
64  
65          //Otsikko
66          JLabel logo = new JLabel(new ImageIcon(ClassLoader.getSystemResource(HYMIOPELILOGO)));
67  
68          //Iso-paneeli & Iso-layout    
69          JPanel iso = new JPanel();
70          iso.setLayout(new BoxLayout(iso, BoxLayout.Y_AXIS));
71          iso.setBackground(new Color(249, 108, 57));
72  
73          //Apuri & Paneelit
74          try {
75              apuri = new Apuri(hymiopeli, this);
76          } catch (IOException e) {
77              System.out.println("Tiedostoja ei löytynyt. Konsultoi iilumme@cs.helsinki.fi ;)");
78              System.exit(0);
79          }
80  
81          //Piirtoalusta
82          this.piirtoalusta = new Piirtoalusta(hymiopeli);
83          piirtoalusta.setVisible(false);
84  
85          //HymioPeli
86          hymiopeli.setApuri(apuri);
87          hymiopeli.setPiirtoalusta(piirtoalusta);
88  
89          //Add 
90          c.add(logo, BorderLayout.NORTH);
91  
92          iso.add(apuri.getKieliPaneeli());
93          iso.add(apuri.getValikkoPaneeli());
94          iso.add(apuri.getHahmoPaneeli());
95          iso.add(piirtoalusta);
96  
97          c.add(iso, BorderLayout.CENTER);
98          c.add(apuri.getTietoPaneeli(), BorderLayout.SOUTH);
99      }
100 
101     private void setKeyListener() {
102         frame.setFocusable(true);
103         NappaimistonKuuntelija nK = new NappaimistonKuuntelija(hymiopeli, piirtoalusta);
104         frame.addKeyListener(nK);
105         hymiopeli.setNappaimistonKuuntelija(nK);
106     }
107 
108 }

1   package iilumme.hymiopeli.ui;
2   
3   import iilumme.hymiopeli.ui.listeners.ValikkoNapinKuuntelija;
4   import iilumme.hymiopeli.util.KieliUtil;
5   import java.awt.Color;
6   import java.awt.GridLayout;
7   import javax.swing.ImageIcon;
8   import javax.swing.JButton;
9   import javax.swing.JPanel;
10  
11  /**
12   * Luokka, joka luo paneelin, jossa on neljä nappia. Napit edustavat eri
13   * valikkovalintoja.
14   */
15  public class ValikkoPanel extends JPanel {
16  
17      private final Apuri apuri;
18  
19      private JButton uusiPeli;
20      private JButton highscore;
21      private JButton ohjeet;
22      private JButton poistu;
23      private static final String POKAALIKUVA = "Images/trophy.png";
24  
25      public ValikkoPanel(Apuri a) {
26          super(new GridLayout(1, 4));
27          this.apuri = a;
28          luoKomponentit();
29          setBackground(new Color(249, 108, 57));
30          asetaPois();
31      }
32  
33      private void luoKomponentit() {
34  
35          uusiPeli = new JButton();
36          highscore = new JButton(new ImageIcon(ClassLoader.getSystemResource(POKAALIKUVA)));
37          ohjeet = new JButton();
38          poistu = new JButton();
39  
40          uusiPeli.addActionListener(new ValikkoNapinKuuntelija(uusiPeli, apuri));
41          highscore.addActionListener(new ValikkoNapinKuuntelija(highscore, apuri));
42          ohjeet.addActionListener(new ValikkoNapinKuuntelija(ohjeet, apuri));
43          poistu.addActionListener(new ValikkoNapinKuuntelija(poistu, apuri));
44  
45          add(uusiPeli);
46          add(highscore);
47          add(ohjeet);
48          add(poistu);
49      }
50  
51      /**
52       * Päivittää napeille oikean kieliset tekstit.
53       *
54       * @see iilumme.hymiopeli.util.KieliUtil#getString(java.lang.String)
55       */
56      public void paivita() {
57          uusiPeli.setText(KieliUtil.getString("uusipeli"));
58          highscore.setText(KieliUtil.getString("highscore"));
59          ohjeet.setText(KieliUtil.getString("ohjeet"));
60          poistu.setText(KieliUtil.getString("poistu"));
61  
62      }
63  
64      /**
65       * Asettaa paneelin näkyväksi sekä toiminnalliseksi.
66       */
67      public void asetaNakyviin() {
68          setEnabled(true);
69          setVisible(true);
70      }
71  
72      /**
73       * Asettaa paneelin näkymättömäksi sekä toimimattomaksi.
74       */
75      public void asetaPois() {
76          setEnabled(false);
77          setVisible(false);
78      }
79  
80  }

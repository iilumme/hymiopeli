3   import iilumme.hymiopeli.ui.Apuri;
4   import iilumme.hymiopeli.util.KieliUtil;
5   import java.awt.event.ActionEvent;
6   import java.awt.event.ActionListener;
7   import javax.swing.JButton;
8   
9   /**
10   * Kuuntelija KieliPanelin napeille.
11   */
12  public class KieliNapinKuuntelija implements ActionListener {
13  
14      private final JButton nappi;
15      private final Apuri apuri;
16  
17      public KieliNapinKuuntelija(JButton nappi, Apuri a) {
18          this.nappi = nappi;
19          this.apuri = a;
20      }
21  
22      /**
23       * Asettaa kieleksi sen minkä käyttäjä on valinnut. Päivitetään
24       * ValikkoPanelin tekstit ja asetetaan se näkyväksi.
25       *
26       * @param ae
27       *
28       * @see iilumme.hymiopeli.ui.Apuri#getKieliPaneeli()
29       * @see iilumme.hymiopeli.ui.Apuri#getValikkoPaneeli()
30       * @see iilumme.hymiopeli.ui.KieliPanel#asetaPois()
31       * @see iilumme.hymiopeli.ui.ValikkoPanel#paivita()
32       * @see iilumme.hymiopeli.ui.ValikkoPanel#asetaNakyviin()
33       */
34      @Override
35      public void actionPerformed(ActionEvent ae) {
36  
37          KieliUtil.setLanguage(nappi.getText());
38          apuri.getKieliPaneeli().asetaPois();
39          apuri.getValikkoPaneeli().paivita();
40          apuri.getValikkoPaneeli().asetaNakyviin();
41  
42      }
43  
44  }

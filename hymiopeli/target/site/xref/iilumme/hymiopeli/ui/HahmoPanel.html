1   package iilumme.hymiopeli.ui;
2   
3   import iilumme.hymiopeli.ui.listeners.HahmoNapinKuuntelija;
4   import java.awt.Color;
5   import java.awt.GridLayout;
6   import javax.swing.ImageIcon;
7   import javax.swing.JButton;
8   import javax.swing.JPanel;
9   
10  /**
11   * Luokka, joka luo paneelin, jossa on kolme nappia. Napit ovat valittavia
12   * hahmoja.
13   */
14  public class HahmoPanel extends JPanel {
15  
16      private final Apuri apuri;
17      private static final String IRONMANLOGO = "Images/ironman.png";
18      private static final String TIIKERILOGO = "Images/tiger.jpg";
19      private static final String HYMIOLOGO = "Images/hymio.jpg";
20  
21      public HahmoPanel(Apuri a) {
22          super(new GridLayout(1, 3));
23          this.apuri = a;
24          luoKomponentit();
25          setBackground(new Color(249, 108, 57));
26          asetaPois();
27      }
28  
29      private void luoKomponentit() {
30  
31          JButton hymio = new JButton(new ImageIcon(ClassLoader.getSystemResource(HYMIOLOGO)));
32          JButton tikru = new JButton(new ImageIcon(ClassLoader.getSystemResource(TIIKERILOGO)));
33          JButton ironman = new JButton(new ImageIcon(ClassLoader.getSystemResource(IRONMANLOGO)));
34  
35          hymio.addActionListener(new HahmoNapinKuuntelija(hymio, apuri));
36          tikru.addActionListener(new HahmoNapinKuuntelija(tikru, apuri));
37          ironman.addActionListener(new HahmoNapinKuuntelija(ironman, apuri));
38  
39          hymio.setName("hy");
40          tikru.setName("ti");
41          ironman.setName("ir");
42  
43          add(hymio);
44          add(tikru);
45          add(ironman);
46      }
47  
48      /**
49       * Asettaa paneelin näkyväksi sekä toiminnalliseksi.
50       */
51      public void asetaNakyviin() {
52          setEnabled(true);
53          setVisible(true);
54      }
55  
56      /**
57       * Asettaa paneelin näkymättömäksi sekä toimimattomaksi.
58       */
59      public void asetaPois() {
60          setEnabled(false);
61          setVisible(false);
62      }
63  }
